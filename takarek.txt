void parseUnit(std::ifstream & f) {
	std::vector<std::string> v;
	std::string sor = "";
	while (!f.eof()) {
		getline(f,sor);
		v.push_back(sor);
	}
	f.close();
	cout << v.size()<< endl;
	for (int i = 12; i < v[1].size() - 2; i++) {
		sor = sor + v[1][i];
	}
	cout << sor << endl;
	sor = "";
	for (int i =9; i < v[2].size() - 1; i++) {
		sor = sor + v[2][i];
	}
	cout << sor << endl;
	sor = "";
	for (int i = 10; i < v[3].size(); i++) {
		sor = sor + v[3][i];
	}
	cout << sor << endl;

}
